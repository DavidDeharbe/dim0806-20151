    \li \For $j = 2$ \To $\id{length}[A]$ \> \> \> \> \> \> \> \> \> \> $c_1$ \> $n$

    \li   $i \gets j-1$ \> \> \> \> \> \> \> \> $c_3$ \> $n-1$
    \li   \While $i > 0 \land A[i] > \id{key}$  \> \> \> \> \> \> \> \> $c_4$ \> $\sum_{j=2}^{n} t_j$
    \li   \Do
    \li     $A[i+1] \gets A[i]$ \> \> \> \> \> \> $c_5$ \> $\sum_{j=2}^{n} (t_j-1)$
    \li     $i \gets i - 1$     \> \> \> \> \> \> $c_6$ \> $\sum_{j=2}^{n} (t_j-1)$
          \End
    \li   $A[i+1] \gets \id{key}$ \> \> \> \> \> \> \> \> $c_7$ \> $n-1$
        \End
